# Responsive Input

<FirstParagraph>
  A responsive utility for creating dynamic inputs based on breakpoints.
</FirstParagraph>

Use this component to expose controls that target breakpoints or a single, global
value. Responsive outputs contain an array of values based on the number of breakpoints
in the theme.

## Example

<ResponsiveExample />

## Usage

```ts
import { useState } from 'react'
import {
  Length,
  ResponsiveLength,
  LengthInput,
  ResponsiveInput,
} from '@compai/css-gui'

export const ResponsiveExample = () => {
  const [value, setValue] = useState<Length | ResponsiveLength>({
    value: 16,
    unit: 'px',
  })

  return (
    <>
      <ResponsiveInput
        label="Font size"
        value={value}
        onChange={setValue}
        Component={LengthInput}
      />
      <pre>{JSON.stringify(value, null, 2)}</pre>
    </>
  )
}
```
